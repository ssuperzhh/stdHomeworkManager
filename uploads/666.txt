SELECT A.*,(SELECT TARGET_NAME FROM PEIS_TARGET_REPOSITORY B WHERE B.TARGET_ID = A.TARGET_ID AND B.TARGET_VER=A.TARGET_VER ) TARGET_NAME,
(SELECT PROJECT_NAME FROM PEIS_PROJECT_REPOSITORY B WHERE B.PROJECT_ID = A.PROJECT_ID AND B.PROJECT_VER=A.PROJECT_VER ) PROJECT_NAME
 FROM PEIS_REGISTER_TARGETS A WHERE A.REG_ID =:RegId
